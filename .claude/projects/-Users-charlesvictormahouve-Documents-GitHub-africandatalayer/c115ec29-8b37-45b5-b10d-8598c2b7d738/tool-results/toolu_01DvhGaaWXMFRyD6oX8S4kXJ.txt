   219→function normalizeIp(raw: string | null): string | null {
   220→  if (!raw) return null;
   221→  const first = raw.split(",")[0]?.trim();
   222→  if (!first) return null;
   223→  const cleaned = first.replace(/^\[|\]$/g, "");
   224→  if (cleaned.startsWith("::ffff:")) return cleaned.slice(7);
   225→  if (/^\d+\.\d+\.\d+\.\d+:\d+$/.test(cleaned)) return cleaned.split(":")[0] ?? null;
   226→  return cleaned;
   227→}

   229→function isPrivateIp(ip: string): boolean {
   230→  if (ip === "127.0.0.1" || ip === "::1") return true;
   231→  if (ip.startsWith("10.")) return true;
   232→  if (ip.startsWith("192.168.")) return true;
   233→  if (ip.startsWith("172.")) {
   234→    const second = Number(ip.split(".")[1]);
   235→    if (second >= 16 && second <= 31) return true;
   236→  }
   237→  if (ip.startsWith("fc") || ip.startsWith("fd")) return true;
   238→  if (ip.startsWith("fe80:")) return true;
   239→  return false;
   240→}

   287→async function fetchIpInfo(ip: string): Promise<IpLookupResult> {
   288→  const target = `https://ipapi.co/${ip}/json/`;
   289→  const controller = new AbortController();
   290→  const timeout = setTimeout(() => controller.abort(), 3000);
   291→  try {
   292→    const res = await fetch(target, { signal: controller.signal });
   293→    if (!res.ok) {
   294→      return {
   295→        ip,
   296→        location: null,
   297→        asn: null,
   298→        org: null,
   299→        isDatacenter: false,
   300→        isVpn: false,
   301→        isTor: false,
   302→      };
   303→    }
   304→    const data = (await res.json()) as IpApiResponse;
   305→    const latitude = Number(data?.latitude);
   306→    const longitude = Number(data?.longitude);
   307→    const org = toLowerText(data?.org) || null;
   308→    const asn = typeof data?.asn === "string" && data.asn.trim() ? data.asn.trim().toLowerCase() : null;
   309→    const classification = classifyIpOrg(org ?? "");
   310→    return {
   311→      ip,
   312→      location: Number.isFinite(latitude) && Number.isFinite(longitude) ? { latitude, longitude } : null,
   313→      asn,
   314→      org,
   315→      isDatacenter: classification.isDatacenter,
   316→      isVpn: classification.isVpn,
   317→      isTor: classification.isTor,
   318→    };
   319→  } catch {
   320→    return {
   321→      ip,
   322→      location: null,
   323→      asn: null,
   324→      org: null,
   325→      isDatacenter: false,
   326→      isVpn: false,
   327→      isTor: false,
   328→    };
   329→  } finally {
   330→    clearTimeout(timeout);
   331→  }
   332→}

   412→async function buildFallbackFraudCheck(
   413→  event: PointEvent,
   414→  existingFraudCheck: SubmissionFraudCheck | null,
   415→): Promise<SubmissionFraudCheck | null> {
   416→  const primaryUrl = typeof event.photoUrl === "string" ? event.photoUrl.trim() : "";
   417→  const secondaryUrl = getSecondaryPhotoUrl(event);
   418→  if (!primaryUrl && !secondaryUrl) return existingFraudCheck;
   419→
   420→  const needsPrimaryRecovery = Boolean(primaryUrl) && isPhotoMetadataEffectivelyEmpty(existingFraudCheck?.primaryPhoto);
   421→  const needsSecondaryRecovery = Boolean(secondaryUrl) && isPhotoMetadataEffectivelyEmpty(existingFraudCheck?.secondaryPhoto);
   422→  const primaryExtracted = needsPrimaryRecovery ? await extractPhotoMetadataFromUrl(primaryUrl) : null;
   423→  const secondaryExtracted = needsSecondaryRecovery ? await extractPhotoMetadataFromUrl(secondaryUrl) : null;
   424→  const recoveredPrimaryPhoto = buildPhotoFraudMetadata({
   425→    extracted: primaryExtracted,
   426→    submissionLocation: event.location,
   427→    ipLocation: null,
   428→    submissionMatchThresholdKm: SUBMISSION_PHOTO_MATCH_KM,
   429→    ipMatchThresholdKm: IP_PHOTO_MATCH_KM,
   430→  });
   431→  const recoveredSecondaryPhoto = buildPhotoFraudMetadata({
   432→    extracted: secondaryExtracted,
   433→    submissionLocation: event.location,
   434→    ipLocation: null,
   435→    submissionMatchThresholdKm: SUBMISSION_PHOTO_MATCH_KM,
   436→    ipMatchThresholdKm: IP_PHOTO_MATCH_KM,
   437→  });
   438→  const primaryPhoto = mergePhotoFraudMetadata(existingFraudCheck?.primaryPhoto ?? null, recoveredPrimaryPhoto, Boolean(primaryUrl));
   439→  const secondaryPhoto = mergePhotoFraudMetadata(
   440→    existingFraudCheck?.secondaryPhoto ?? null,
   441→    recoveredSecondaryPhoto,
   442→    Boolean(secondaryUrl),
   443→  );
   444→  if (!primaryPhoto && !secondaryPhoto) return null;
   445→
   446→  return buildSubmissionFraudCheck({
   447→    submissionLocation: existingFraudCheck?.submissionLocation ?? event.location,
   448→    effectiveLocation: existingFraudCheck?.effectiveLocation ?? event.location,
   449→    ipLocation: existingFraudCheck?.ipLocation ?? null,
   450→    primaryPhoto,
   451→    secondaryPhoto,
   452→    submissionMatchThresholdKm: existingFraudCheck?.submissionMatchThresholdKm ?? SUBMISSION_PHOTO_MATCH_KM,
   453→    ipMatchThresholdKm: existingFraudCheck?.ipMatchThresholdKm ?? IP_PHOTO_MATCH_KM,
   454→  });
   455→}

   457→async function buildAdminSubmissionEvents(events: PointEvent[]): Promise<AdminSubmissionEvent[]> {
   458→  const sortedEvents = [...events].sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
   459→  const userIds = Array.from(
   460→    new Set(
   461→      sortedEvents
   462→        .map((event) => normalizeActorId(event.userId))
   463→        .filter((value) => value.length > 0),
   464→    ),
   465→  );
   466→
   467→  const profileEntries = await Promise.all(
   468→    userIds.map(async (userId) => [userId, await getUserProfile(userId)] as const),
   469→  );
   470→  const profileByUserId = new Map(profileEntries);
   471→  const output: AdminSubmissionEvent[] = [];
   472→  let fallbackLookups = 0;
   473→
   474→  for (const event of sortedEvents) {
   475→    const normalizedUserId = normalizeActorId(event.userId);
   476→    const profile = profileByUserId.get(normalizedUserId) ?? null;
   477→    const emailFromProfile = typeof profile?.email === "string" ? profile.email.trim().toLowerCase() : "";
   478→    const email = emailFromProfile || (normalizedUserId.includes("@") ? normalizedUserId : null);
   479→    const name = getUserDisplayName(normalizedUserId, profile?.name ?? null, email);
   480→    let fraudCheck = getEventFraudCheck(event);
   481→    if (shouldRunFallbackFraudCheck(event, fraudCheck) && fallbackLookups < MAX_FORENSICS_FALLBACK_LOOKUPS) {
   482→      fallbackLookups += 1;
   483→      try {
   484→        fraudCheck = await buildFallbackFraudCheck(event, fraudCheck);
   485→      } catch {
   486→        fraudCheck = fraudCheck ?? null;
   487→      }
   488→    }
   489→
   490→    output.push({
   491→      event,
   492→      user: {
   493→        id: normalizedUserId || event.userId,
   494→        name,
   495→        email,
   496→      },
   497→      fraudCheck,
   498→    });
   499→  }
   500→
   501→  return output;
   502→}

   504→async function buildCombinedEvents(): Promise<PointEvent[]> {
   505→  const pointEvents = (await getPointEvents()).map(stripInlinePhotoData);
   506→  const legacySubmissions = await getLegacySubmissions();
   507→  const merged = mergePointEventsWithLegacy(pointEvents, legacySubmissions);
   508→  const seenExternalIds = new Set(
   509→    merged
   510→      .map((event) => (typeof event.externalId === "string" ? event.externalId.trim() : ""))
   511→      .filter((value) => value.length > 0),
   512→  );
   513→  const seenPointIds = new Set(merged.map((event) => event.pointId));
   514→  for (const seedEvent of BONAMOUSSADI_CURATED_SEED_EVENTS) {
   515→    const externalId = typeof seedEvent.externalId === "string" ? seedEvent.externalId.trim() : "";
   516→    if (externalId && seenExternalIds.has(externalId)) continue;
   517→    if (seenPointIds.has(seedEvent.pointId)) continue;
   518→    merged.push(seedEvent);
   519→    if (externalId) seenExternalIds.add(externalId);
   520→    seenPointIds.add(seedEvent.pointId);
   521→  }
   522→  return merged;
   523→}

   537→    expected: EXPECTED_SUBMISSION_CATEGORIES,
   538→    actual: [],
   539→    missing: EXPECTED_SUBMISSION_CATEGORIES,
   540→    extra: [],
   541→  };
   542→
   543→  try {
   544→    const result = await query<{ constraint_definition: string | null }>(
   545→      `
   546→        SELECT pg_get_constraintdef(c.oid) AS constraint_definition
   547→        FROM pg_constraint c
   548→        JOIN pg_class rel ON rel.oid = c.conrelid
   549→        JOIN pg_namespace ns ON ns.oid = rel.relnamespace
   550→        WHERE ns.nspname = 'public'
   551→          AND rel.relname = 'point_events'
   552→          AND c.conname = 'point_events_category_check'
   553→        LIMIT 1
   554→      `,
   555→    );
   556→    const definition = result.rows[0]?.constraint_definition ?? null;
   557→    if (!definition) {
   558→      return { ...fallback, reason: "constraint_not_found" };
   559→    }
   560→    const actual = parseConstraintCategories(definition);
   561→    const diff = diffCategorySets(EXPECTED_SUBMISSION_CATEGORIES, actual);
   562→    return {
   563→      ok: diff.ok,
   564→      expected: diff.expected,
   565→      actual: diff.actual,
   566→      missing: diff.missing,
   567→      extra: diff.extra,
   568→    };
   569→  } catch (error) {
   570→    if (isStorageUnavailableError(error)) {
   571→      return { ...fallback, reason: "storage_unavailable" };
   572→    }
   573→    return { ...fallback, reason: "query_failed" };
   574→  }
   575→}

   577→export async function GET(request: Request): Promise<Response> {
   578→  const auth = await requireUser(request);
   579→  const authContext = toSubmissionAuthContext(auth);
   580→  const url = new URL(request.url);
   581→  const view = url.searchParams.get("view");
   582→  const lat = url.searchParams.get("lat");
   583→  const lng = url.searchParams.get("lng");
   584→  const radius = url.searchParams.get("radius");
   585→  const requestedScope = normalizeMapScope(url.searchParams.get("scope"));
   586→  const canUseExpandedScope = requestedScope !== "bonamoussadi";
   587→  if (canUseExpandedScope && !authContext) return errorResponse("Unauthorized", 401);
   588→  if (canUseExpandedScope && !authContext.isAdmin) return errorResponse("Forbidden", 403);
   589→  const effectiveScope = canUseExpandedScope ? requestedScope : "bonamoussadi";
   590→  const canUsePublicCache = !authContext && !canUseExpandedScope;
   591→
   592→  try {
   593→    if (view === "schema_guard") {
   594→      const adminAccess = resolveAdminViewAccess(authContext);
   595→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   596→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   597→      return jsonResponse(await buildSchemaGuardView(), { status: 200 });
   598→    }
   599→
   600→    const allEvents = await buildCombinedEvents();
   601→    const scopedEvents = allEvents.filter((event) => {
   602→      if (effectiveScope === "global") return true;
   603→      if (effectiveScope === "cameroon") return isWithinCameroon(event.location);
   604→      return isWithinBonamoussadi(event.location);
   605→    });
   606→
   607→    if (view === "events") {
   608→      if (!authContext) {
   609→        return jsonResponse(redactEventUserIds(scopedEvents), {
   610→          status: 200,
   611→          headers: {
   612→            "cache-control": PUBLIC_READ_CACHE_CONTROL,
   613→          },
   614→        });
   615→      }
   616→      const responseEvents = filterEventsForViewer(scopedEvents, authContext);
   617→      return jsonResponse(responseEvents, { status: 200 });
   618→    }
   619→
   620→    if (view === "admin_events") {
   621→      const adminAccess = resolveAdminViewAccess(authContext);
   622→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   623→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   624→      const adminEvents = await buildAdminSubmissionEvents(scopedEvents);
   625→      return jsonResponse(adminEvents, { status: 200 });
   626→    }
   627→
   628→    if (view === "review_queue") {
   629→      const adminAccess = resolveAdminViewAccess(authContext);
   630→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   631→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   632→      const pending = scopedEvents.filter((event) => isPendingReview(event));
   633→      const adminEvents = await buildAdminSubmissionEvents(pending);
   634→      return jsonResponse(adminEvents, { status: 200 });
   635→    }
   636→
   637→    let projected = projectPointsFromEvents(scopedEvents);
   638→
   639→    if (view === "dedup_candidates") {
   640→      if (!authContext) return errorResponse("Unauthorized", 401);
   641→      const category = normalizeCategory(url.searchParams.get("category") ?? undefined);
   642→      if (!category) return errorResponse("Invalid category", 400);
   643→      const latitude = Number(url.searchParams.get("lat"));
   644→      const longitude = Number(url.searchParams.get("lng"));
   645→      if (!Number.isFinite(latitude) || !Number.isFinite(longitude)) {
   646→        return errorResponse("lat and lng are required", 400);
   647→      }
   648→      const name = trimString(url.searchParams.get("name"), 160);
   649→      const probeDetails: SubmissionDetails = {};
   650→      if (name) {
   651→        probeDetails.name = name;
   652→        probeDetails.siteName = name;
   653→        probeDetails.roadName = name;
   654→      }
   655→      const result = buildDedupCandidates(
   656→        category,
   657→        { latitude, longitude },
   658→        probeDetails,
   659→        projected,
   660→      );
   661→      return jsonResponse(result);
   662→    }
   663→
   664→    if (lat && lng && radius) {
   665→      const latitude = Number(lat);
   666→      const longitude = Number(lng);
   667→      const radiusKm = Number(radius);
   668→      if (Number.isFinite(latitude) && Number.isFinite(longitude) && Number.isFinite(radiusKm)) {
   669→        projected = projected.filter((point) => haversineKm(point.location, { latitude, longitude }) <= radiusKm);
   670→      }
   671→    }
   672→
   673→    return jsonResponse(projected, {
   674→      status: 200,
   675→      headers: canUsePublicCache
   676→        ? {
   677→            "cache-control": PUBLIC_READ_CACHE_CONTROL,
   678→          }
   679→        : undefined,
   680→    });
   681→  } catch (error) {
   682→    if (isStorageUnavailableError(error)) {
   683→      return errorResponse("Storage service temporarily unavailable", 503, { code: "storage_unavailable" });
   684→    }
   685→    throw error;
   686→  }

   689→export async function POST(request: Request): Promise<Response> {

   693→  const isAdminUser = authContext?.isAdmin === true;
   694→

   696→  try {
   697→    body = (await request.json()) as SubmissionInput;
   698→  } catch {
   699→    return errorResponse("Invalid JSON body", 400);
   700→  }
   701→
   702→  const category = normalizeCategory(body?.category as string | undefined);
   703→  if (!category) {
   704→    return errorResponse("Invalid category", 400);
   705→  }
   706→
   707→  const eventType = normalizeEventType(body?.eventType);

   709→  const dedupDecision = normalizeDedupDecision(rawBody?.dedupDecision);

   711→    typeof rawBody?.dedupTargetPointId === "string"
   712→      ? (rawBody.dedupTargetPointId as string).trim()
   713→      : "";

   715→  const details = normalizeEnrichPayload(
   716→    category,
   717→    body?.details && typeof body.details === "object" ? ({ ...(body.details as SubmissionDetails) } as SubmissionDetails) : {},
   718→  );
   719→  const requestUserAgent = trimString(request.headers.get("user-agent"), 256);
   720→  const clientDevice = sanitizeClientDevice((details as Record<string, unknown>).clientDevice);
   721→  if (clientDevice) {
   722→    if (!clientDevice.userAgent && requestUserAgent) clientDevice.userAgent = requestUserAgent;
   723→    details.clientDevice = clientDevice;
   724→  } else if ("clientDevice" in details) {
   725→    delete details.clientDevice;
   726→  }
   727→

   729→  if (!imageBase64) return errorResponse("Photo is required", 400);

  1058→    details.riskScore = riskEvaluation.riskScore;
  1059→    details.riskComponents = riskEvaluation.riskComponents as unknown as Record<string, unknown>;
  1060→    details.exifTrustScore = riskEvaluation.exifTrustScore;
  1061→    details.velocitySignals = riskEvaluation.velocity as unknown as Record<string, unknown>;
  1062→    details.ipHash = ipHash ?? undefined;
  1063→    if (ipInfo) {
  1064→      details.ipReputation = {
  1065→        asn: ipInfo.asn,
  1066→        org: ipInfo.org,
  1067→        isDatacenter: ipInfo.isDatacenter,
