    72→    .map((value) => (typeof value === 'string' ? value.trim() : ''))

    90→function dataUrlToBytes(dataUrl: string): Uint8Array | null {
    91→  const commaIndex = dataUrl.indexOf(',');
    92→  if (commaIndex === -1) return null;
    93→  const base64 = dataUrl.slice(commaIndex + 1);
    94→  if (!base64) return null;
    95→  const binary = atob(base64);
    96→  const bytes = new Uint8Array(binary.length);
    97→  for (let i = 0; i < binary.length; i += 1) {
    98→    bytes[i] = binary.charCodeAt(i);
    99→  }
   100→  return bytes;
   101→}

   103→function bytesToDataUrl(bytes: Uint8Array, mime: string): string {
   104→  let binary = '';
   105→  for (let i = 0; i < bytes.length; i += 1) {
   106→    binary += String.fromCharCode(bytes[i] ?? 0);
   107→  }
   108→  return `data:${mime};base64,${btoa(binary)}`;
   109→}

   111→function extractJpegExifSegment(bytes: Uint8Array): Uint8Array | null {
   112→  if (bytes.length < 4) return null;
   113→  if (bytes[0] !== 0xff || bytes[1] !== 0xd8) return null;
   114→
   115→  let offset = 2;
   116→  while (offset + 4 <= bytes.length) {
   117→    if (bytes[offset] !== 0xff) {
   118→      offset += 1;
   119→      continue;
   120→    }
   121→
   122→    const marker = bytes[offset + 1];
   123→    if (marker === 0xd8 || marker === 0x01 || (marker >= 0xd0 && marker <= 0xd7)) {
   124→      offset += 2;
   125→      continue;
   126→    }
   127→    if (marker === 0xda || marker === 0xd9) break;
   128→
   129→    const segmentLength = ((bytes[offset + 2] ?? 0) << 8) | (bytes[offset + 3] ?? 0);
   130→    if (segmentLength < 2) break;
   131→    const segmentEnd = offset + 2 + segmentLength;
   132→    if (segmentEnd > bytes.length) break;
   133→
   134→    if (marker === 0xe1 && segmentLength >= 8) {
   135→      const signature = String.fromCharCode(
   136→        bytes[offset + 4] ?? 0,
   137→        bytes[offset + 5] ?? 0,
   138→        bytes[offset + 6] ?? 0,
   139→        bytes[offset + 7] ?? 0,
   140→        bytes[offset + 8] ?? 0,
   141→        bytes[offset + 9] ?? 0,
   142→      );
   143→      if (signature === 'Exif\0\0') {
   144→        return bytes.slice(offset, segmentEnd);
   145→      }
   146→    }
   147→
   148→    offset = segmentEnd;
   149→  }
   150→
   151→  return null;
   152→}

   154→function injectExifIntoJpeg(compressedBytes: Uint8Array, exifSegment: Uint8Array): Uint8Array | null {
   155→  if (compressedBytes.length < 2) return null;
   156→  if (compressedBytes[0] !== 0xff || compressedBytes[1] !== 0xd8) return null;
   157→  if (!exifSegment.length) return null;
   158→
   159→  const merged = new Uint8Array(compressedBytes.length + exifSegment.length);
   160→  merged[0] = compressedBytes[0];
   161→  merged[1] = compressedBytes[1];
   162→  merged.set(exifSegment, 2);
   163→  merged.set(compressedBytes.slice(2), 2 + exifSegment.length);
   164→  return merged;
   165→}

   167→async function preserveJpegExif(originalFile: File, compressedDataUrl: string): Promise<string> {
   168→  if (!JPEG_MIME_TYPES.has((originalFile.type || '').toLowerCase())) return compressedDataUrl;
   169→
   170→  const originalBytes = new Uint8Array(await originalFile.arrayBuffer());
   171→  const exifSegment = extractJpegExifSegment(originalBytes);
   172→  if (!exifSegment) return compressedDataUrl;
   173→
   174→  const compressedBytes = dataUrlToBytes(compressedDataUrl);
   175→  if (!compressedBytes) return compressedDataUrl;
   176→  if (extractJpegExifSegment(compressedBytes)) return compressedDataUrl;
   177→
   178→  const merged = injectExifIntoJpeg(compressedBytes, exifSegment);
   179→  if (!merged) return compressedDataUrl;
   180→  return bytesToDataUrl(merged, 'image/jpeg');
   181→}

   183→function parseDeviceFromUserAgent(): { make: string | null; model: string | null } {
   184→  const ua = navigator.userAgent;
   185→  // iPhone detection
   186→  const iphoneMatch = ua.match(/iPhone/);
   187→  if (iphoneMatch) return { make: 'Apple', model: 'iPhone' };
   188→  // iPad detection
   189→  const ipadMatch = ua.match(/iPad/);
   190→  if (ipadMatch) return { make: 'Apple', model: 'iPad' };
   191→  // Android device detection
   192→  const androidMatch = ua.match(/;\s*([^;)]+)\s+Build\//);
   193→  if (androidMatch) return { make: null, model: androidMatch[1]?.trim() ?? null };
   194→  return { make: null, model: null };
   195→}

   197→async function extractClientExif(
   198→  file: File,
   199→  deviceLocation: { latitude: number; longitude: number } | null,
   200→): Promise<ClientExifData | null> {
   201→  // Step 1: Try extracting EXIF from the raw file (works on Android, some iOS versions)
   202→  let lat: number | null = null;
   203→  let lng: number | null = null;
   204→  let capturedAt: string | null = null;
   205→  let deviceMake: string | null = null;
   206→  let deviceModel: string | null = null;
   207→
   208→  try {
   209→    const buffer = await file.arrayBuffer();
   210→    const [exifData, gpsData] = await Promise.all([
   211→      exifr.parse(buffer, { gps: true, exif: true, tiff: true, pick: [
   212→        'latitude', 'longitude', 'GPSLatitude', 'GPSLongitude',
   213→        'DateTimeOriginal', 'DateTimeDigitized', 'CreateDate', 'ModifyDate',
   214→        'Make', 'Model',
   215→      ] }).catch(() => null),
   216→      exifr.gps(buffer).catch(() => null),
   217→    ]);
   218→    const merged = { ...(exifData ?? {}), ...(gpsData ?? {}) } as Record<string, unknown>;
   219→    lat = typeof merged.latitude === 'number' ? merged.latitude : null;
   220→    lng = typeof merged.longitude === 'number' ? merged.longitude : null;
   221→    const rawDate = merged.DateTimeOriginal ?? merged.DateTimeDigitized ?? merged.CreateDate ?? merged.ModifyDate;
   222→    if (rawDate instanceof Date) {
   223→      capturedAt = rawDate.toISOString();
   224→    } else if (typeof rawDate === 'string' && rawDate.trim()) {
   225→      const m = rawDate.match(/^(\d{4}):(\d{2}):(\d{2})\s+(\d{2}):(\d{2}):(\d{2})/);
   226→      capturedAt = m ? `${m[1]}-${m[2]}-${m[3]}T${m[4]}:${m[5]}:${m[6]}` : rawDate;
   227→    }
   228→    deviceMake = typeof merged.Make === 'string' ? merged.Make : null;
   229→    deviceModel = typeof merged.Model === 'string' ? merged.Model : null;
   230→  } catch {
   231→    // EXIF parsing failed entirely -- fall through to device fallback
   232→  }
   233→
   234→  // Step 2: Fill gaps with browser APIs (critical for iOS which strips EXIF from <input capture>)
   235→  if (deviceLocation) {
   236→    if (lat == null) lat = deviceLocation.latitude;
   237→    if (lng == null) lng = deviceLocation.longitude;
   238→  }
   239→  if (!capturedAt) {
   240→    capturedAt = new Date().toISOString();
   241→  }
   242→  if (!deviceMake && !deviceModel) {
   243→    const parsed = parseDeviceFromUserAgent();
   244→    deviceMake = parsed.make;
   245→    deviceModel = parsed.model;
   246→  }
   247→
   248→  if (lat == null && lng == null && !capturedAt && !deviceMake && !deviceModel) return null;
   249→  return { latitude: lat, longitude: lng, capturedAt, deviceMake, deviceModel };
   250→}

   563→  const fileToBase64 = async (file: File) => {
   564→    const readAsDataUrl = (target: Blob) =>
   565→      new Promise<string>((resolve, reject) => {
   566→        const reader = new FileReader();
   567→        reader.onload = () => {
   568→          if (typeof reader.result === 'string') {
   569→            resolve(reader.result);
   570→            return;
   571→          }
   572→          reject(new Error('Unable to read file.'));
   573→        };
   574→        reader.onerror = () => reject(new Error('Unable to read file.'));
   575→        reader.readAsDataURL(target);
   576→      });
   577→
   578→    if (!file.type.startsWith('image/')) {
   579→      return await readAsDataUrl(file);
   580→    }
   581→
   582→    // Keep original capture when possible so EXIF (GPS/time/device) reaches server-side forensics.
   583→    if (file.size <= MAX_SUBMISSION_IMAGE_BYTES) {
   584→      return await readAsDataUrl(file);
   585→    }
   586→
   587→    const loadImage = (objectUrl: string) =>
   588→      new Promise<HTMLImageElement>((resolve, reject) => {
   589→        const img = new Image();
   590→        img.onload = () => resolve(img);
   591→        img.onerror = () => reject(new Error('Unable to decode image.'));
   592→        img.src = objectUrl;
   593→      });
   594→
   595→    const objectUrl = URL.createObjectURL(file);
   596→    try {
   597→      const img = await loadImage(objectUrl);
   598→      const maxDimension = Math.max(img.width, img.height);
   599→      const scale = maxDimension > MAX_UPLOAD_DIMENSION ? MAX_UPLOAD_DIMENSION / maxDimension : 1;
   600→      const width = Math.max(1, Math.round(img.width * scale));
   601→      const height = Math.max(1, Math.round(img.height * scale));
   602→      const canvas = document.createElement('canvas');
   603→      canvas.width = width;
   604→      canvas.height = height;
   605→      const ctx = canvas.getContext('2d');
   606→      if (!ctx) {
   607→        return await readAsDataUrl(file);
   608→      }
   609→      ctx.drawImage(img, 0, 0, width, height);
   610→      const quality = isLowEndDevice ? IMAGE_QUALITY_LOW_END : IMAGE_QUALITY_DEFAULT;
   611→      const compressed = canvas.toDataURL('image/jpeg', quality);
   612→      return await preserveJpegExif(file, compressed);
   613→    } catch {
   614→      return await readAsDataUrl(file);
   615→    } finally {
   616→      URL.revokeObjectURL(objectUrl);
   617→    }

   698→  const buildEnrichDetails = (): Record<string, unknown> => {
   699→    const details: Record<string, unknown> = { clientDevice: getClientDeviceInfo() };
   700→    for (const gap of gaps) {
   701→      const fieldConfig = ENRICH_FIELD_CATALOG[gap];
   702→      if (!fieldConfig) {
   703→        const fallback = normalizeText(getEnrichFieldValue(gap));
   704→        if (fallback) details[gap] = fallback;
   705→        continue;
   706→      }
   707→
   708→      if (fieldConfig.kind === 'map_value') {
   709→        if (!enrichTouched[gap]) continue;
   710→        if (gap === 'merchantIdByProvider') {
   711→          const normalizedMerchantId = merchantId.trim();
   712→          if (normalizedMerchantId) details.merchantIdByProvider = { [merchantProvider]: normalizedMerchantId };
   713→          continue;
   714→        }
   715→        if (gap === 'pricesByFuel') {
   716→          const parsedPrice = normalizeNumber(priceValue);
   717→          if (parsedPrice !== null) details.pricesByFuel = { [priceFuelType]: parsedPrice };
   718→          continue;
   719→        }
   720→        const rawMap = getEnrichFieldValue(gap);
   721→        if (rawMap && typeof rawMap === 'object' && !Array.isArray(rawMap)) {
   722→          const normalizedMap = Object.fromEntries(
   723→            Object.entries(rawMap as Record<string, unknown>)
   724→              .map(([key, value]) => [key.trim(), normalizeText(value)])
   725→              .filter(([key, value]) => key.length > 0 && Boolean(value)),
   726→          );
   727→          if (Object.keys(normalizedMap).length > 0) details[gap] = normalizedMap;
   728→        }
   729→        continue;
   730→      }
   731→
   732→      const rawValue = getEnrichFieldValue(gap);
   733→      if (fieldConfig.kind === 'boolean') {
   734→        if (enrichTouched[gap] && typeof rawValue === 'boolean') details[gap] = rawValue;
   735→        continue;
   736→      }
   737→      if (fieldConfig.kind === 'text') {
   738→        const normalized = normalizeText(rawValue);
   739→        if (normalized) details[gap] = normalized;
   740→        continue;
   741→      }
   742→      if (fieldConfig.kind === 'number') {
   743→        const normalized = normalizeNumber(rawValue);
   744→        if (normalized !== null) details[gap] = normalized;
   745→        continue;
   746→      }
   747→      if (fieldConfig.kind === 'single_select') {
   748→        const normalized = normalizeText(rawValue);
   749→        if (normalized) details[gap] = normalized;
   750→        continue;
   751→      }
   752→      if (fieldConfig.kind === 'multi_select') {
   753→        const normalized = normalizeStringList(rawValue);
   754→        if (normalized.length > 0) details[gap] = normalized;
   755→      }
   756→    }
