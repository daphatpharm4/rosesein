   231→function groupEventsByPoint(items: AdminSubmissionEvent[], language: 'en' | 'fr'): GroupedPoint[] {
   232→  const groups = new Map<string, AdminSubmissionEvent[]>();
   233→  for (const item of items) {
   234→    const pid = item.event.pointId;
   235→    const existing = groups.get(pid);
   236→    if (existing) {
   237→      existing.push(item);
   238→    } else {
   239→      groups.set(pid, [item]);
   240→    }
   241→  }
   242→  const result: GroupedPoint[] = [];
   243→  for (const [pointId, events] of groups) {
   244→    const sorted = [...events].sort((a, b) => new Date(a.event.createdAt).getTime() - new Date(b.event.createdAt).getTime());
   245→    const latestEvent = sorted[sorted.length - 1]!;
   246→    const createdEvent = sorted.find((e) => e.event.eventType === 'CREATE_EVENT') ?? null;
   247→    const enrichEvents = sorted.filter((e) => e.event.eventType === 'ENRICH_EVENT');
   248→    const allPhotos: GroupedPoint['allPhotos'] = [];
   249→    for (const ev of sorted) {
   250→      const photoUrl = ev.event.photoUrl;
   251→      if (photoUrl && typeof photoUrl === 'string' && photoUrl.trim()) {
   252→        allPhotos.push({
   253→          url: photoUrl,
   254→          eventType: ev.event.eventType,
   255→          createdAt: ev.event.createdAt,
   256→          metadata: ev.fraudCheck?.primaryPhoto ?? null,
   257→        });
   258→      }
   259→      const details = ev.event.details as SubmissionDetails;
   260→      const secondUrl = typeof details.secondPhotoUrl === 'string' && details.secondPhotoUrl.trim() ? details.secondPhotoUrl : null;
   261→      if (secondUrl) {
   262→        allPhotos.push({
   263→          url: secondUrl,
   264→          eventType: ev.event.eventType + ' (secondary)',
   265→          createdAt: ev.event.createdAt,
   266→          metadata: ev.fraudCheck?.secondaryPhoto ?? null,
   267→        });
   268→      }
   269→    }
   270→    result.push({
   271→      pointId,
   272→      events: sorted,
   273→      category: latestEvent.event.category,
   274→      siteName: getSiteName(createdEvent ?? latestEvent, language),
   275→      latestEvent,
   276→      createdEvent,
   277→      enrichEvents,
   278→      allPhotos,
   279→    });
   280→  }
   281→  return result.sort((a, b) => new Date(b.latestEvent.event.createdAt).getTime() - new Date(a.latestEvent.event.createdAt).getTime());
   282→}
