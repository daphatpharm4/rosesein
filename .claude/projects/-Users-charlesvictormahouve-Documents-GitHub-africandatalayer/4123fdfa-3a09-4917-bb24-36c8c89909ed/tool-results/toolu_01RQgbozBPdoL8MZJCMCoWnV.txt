    25→function exifStatusLabel(status: SubmissionPhotoMetadata['exifStatus'] | null | undefined, language: 'en' | 'fr'): string {
    26→  if (status === 'ok') return language === 'fr' ? 'EXIF present' : 'EXIF present';
    27→  if (status === 'fallback_recovered') return language === 'fr' ? 'Recupere via fallback' : 'Recovered via fallback';
    28→  if (status === 'missing') return language === 'fr' ? 'EXIF absent' : 'EXIF missing';
    29→  if (status === 'unsupported_format') return language === 'fr' ? 'Format non supporte' : 'Unsupported format';
    30→  if (status === 'parse_error') return language === 'fr' ? 'Erreur de lecture EXIF' : 'EXIF parse error';
    31→  return language === 'fr' ? 'Indisponible' : 'Unavailable';

   210→function groupEventsByPoint(items: AdminSubmissionEvent[], language: 'en' | 'fr'): GroupedPoint[] {
   211→  const groups = new Map<string, AdminSubmissionEvent[]>();
   212→  for (const item of items) {
   213→    const pid = item.event.pointId;
   214→    const existing = groups.get(pid);
   215→    if (existing) {
   216→      existing.push(item);
   217→    } else {
   218→      groups.set(pid, [item]);
   219→    }
   220→  }
   221→  const result: GroupedPoint[] = [];
   222→  for (const [pointId, events] of groups) {
   223→    const sorted = [...events].sort((a, b) => new Date(a.event.createdAt).getTime() - new Date(b.event.createdAt).getTime());
   224→    const latestEvent = sorted[sorted.length - 1]!;
   225→    const createdEvent = sorted.find((e) => e.event.eventType === 'CREATE_EVENT') ?? null;
   226→    const enrichEvents = sorted.filter((e) => e.event.eventType === 'ENRICH_EVENT');
   227→    const allPhotos: GroupedPoint['allPhotos'] = [];
   228→    for (const ev of sorted) {
   229→      const photoUrl = ev.event.photoUrl;
   230→      if (photoUrl && typeof photoUrl === 'string' && photoUrl.trim()) {
   231→        allPhotos.push({
   232→          url: photoUrl,
   233→          eventType: ev.event.eventType,
   234→          createdAt: ev.event.createdAt,
   235→          metadata: ev.fraudCheck?.primaryPhoto ?? null,
   236→        });
   237→      }
   238→      const details = ev.event.details as SubmissionDetails;
   239→      const secondUrl = typeof details.secondPhotoUrl === 'string' && details.secondPhotoUrl.trim() ? details.secondPhotoUrl : null;
   240→      if (secondUrl) {
   241→        allPhotos.push({
   242→          url: secondUrl,
   243→          eventType: ev.event.eventType + ' (secondary)',
   244→          createdAt: ev.event.createdAt,
   245→          metadata: ev.fraudCheck?.secondaryPhoto ?? null,
   246→        });
   247→      }
   248→    }
   249→    result.push({
   250→      pointId,
   251→      events: sorted,
   252→      category: latestEvent.event.category,
   253→      siteName: getSiteName(createdEvent ?? latestEvent, language),
   254→      latestEvent,
   255→      createdEvent,
   256→      enrichEvents,
   257→      allPhotos,
   258→    });
   259→  }
   260→  return result.sort((a, b) => new Date(b.latestEvent.event.createdAt).getTime() - new Date(a.latestEvent.event.createdAt).getTime());
   261→}
