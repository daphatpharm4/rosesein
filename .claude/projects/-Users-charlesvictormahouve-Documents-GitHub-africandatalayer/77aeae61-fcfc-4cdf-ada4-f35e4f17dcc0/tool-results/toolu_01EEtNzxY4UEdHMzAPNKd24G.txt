    58→    .map((value) => (typeof value === 'string' ? value.trim() : ''))

    60→  return Array.from(new Set(cleaned));

    76→function dataUrlToBytes(dataUrl: string): Uint8Array | null {
    77→  const commaIndex = dataUrl.indexOf(',');
    78→  if (commaIndex === -1) return null;
    79→  const base64 = dataUrl.slice(commaIndex + 1);
    80→  if (!base64) return null;
    81→  const binary = atob(base64);
    82→  const bytes = new Uint8Array(binary.length);
    83→  for (let i = 0; i < binary.length; i += 1) {
    84→    bytes[i] = binary.charCodeAt(i);
    85→  }
    86→  return bytes;
    87→}

    89→function bytesToDataUrl(bytes: Uint8Array, mime: string): string {
    90→  let binary = '';
    91→  for (let i = 0; i < bytes.length; i += 1) {
    92→    binary += String.fromCharCode(bytes[i] ?? 0);
    93→  }
    94→  return `data:${mime};base64,${btoa(binary)}`;
    95→}

    97→function extractJpegExifSegment(bytes: Uint8Array): Uint8Array | null {
    98→  if (bytes.length < 4) return null;
    99→  if (bytes[0] !== 0xff || bytes[1] !== 0xd8) return null;
   100→
   101→  let offset = 2;
   102→  while (offset + 4 <= bytes.length) {
   103→    if (bytes[offset] !== 0xff) {
   104→      offset += 1;
   105→      continue;
   106→    }
   107→
   108→    const marker = bytes[offset + 1];
   109→    if (marker === 0xd8 || marker === 0x01 || (marker >= 0xd0 && marker <= 0xd7)) {
   110→      offset += 2;
   111→      continue;
   112→    }
   113→    if (marker === 0xda || marker === 0xd9) break;
   114→
   115→    const segmentLength = ((bytes[offset + 2] ?? 0) << 8) | (bytes[offset + 3] ?? 0);
   116→    if (segmentLength < 2) break;
   117→    const segmentEnd = offset + 2 + segmentLength;
   118→    if (segmentEnd > bytes.length) break;
   119→
   120→    if (marker === 0xe1 && segmentLength >= 8) {
   121→      const signature = String.fromCharCode(
   122→        bytes[offset + 4] ?? 0,
   123→        bytes[offset + 5] ?? 0,
   124→        bytes[offset + 6] ?? 0,
   125→        bytes[offset + 7] ?? 0,
   126→        bytes[offset + 8] ?? 0,
   127→        bytes[offset + 9] ?? 0,
   128→      );
   129→      if (signature === 'Exif\0\0') {
   130→        return bytes.slice(offset, segmentEnd);
   131→      }
   132→    }
   133→
   134→    offset = segmentEnd;
   135→  }
   136→
   137→  return null;
   138→}

   140→function injectExifIntoJpeg(compressedBytes: Uint8Array, exifSegment: Uint8Array): Uint8Array | null {
   141→  if (compressedBytes.length < 2) return null;
   142→  if (compressedBytes[0] !== 0xff || compressedBytes[1] !== 0xd8) return null;
   143→  if (!exifSegment.length) return null;
   144→
   145→  const merged = new Uint8Array(compressedBytes.length + exifSegment.length);
   146→  merged[0] = compressedBytes[0];
   147→  merged[1] = compressedBytes[1];
   148→  merged.set(exifSegment, 2);
   149→  merged.set(compressedBytes.slice(2), 2 + exifSegment.length);
   150→  return merged;
   151→}

   153→async function preserveJpegExif(originalFile: File, compressedDataUrl: string): Promise<string> {
   154→  if (!JPEG_MIME_TYPES.has((originalFile.type || '').toLowerCase())) return compressedDataUrl;
   155→
   156→  const originalBytes = new Uint8Array(await originalFile.arrayBuffer());
   157→  const exifSegment = extractJpegExifSegment(originalBytes);
   158→  if (!exifSegment) return compressedDataUrl;
   159→
   160→  const compressedBytes = dataUrlToBytes(compressedDataUrl);
   161→  if (!compressedBytes) return compressedDataUrl;
   162→  if (extractJpegExifSegment(compressedBytes)) return compressedDataUrl;
   163→
   164→  const merged = injectExifIntoJpeg(compressedBytes, exifSegment);
   165→  if (!merged) return compressedDataUrl;
   166→  return bytesToDataUrl(merged, 'image/jpeg');
   167→}

   169→function parseDeviceFromUserAgent(): { make: string | null; model: string | null } {
   170→  const ua = navigator.userAgent;
   171→  // iPhone detection
   172→  const iphoneMatch = ua.match(/iPhone/);
   173→  if (iphoneMatch) return { make: 'Apple', model: 'iPhone' };
   174→  // iPad detection
   175→  const ipadMatch = ua.match(/iPad/);
   176→  if (ipadMatch) return { make: 'Apple', model: 'iPad' };
   177→  // Android device detection
   178→  const androidMatch = ua.match(/;\s*([^;)]+)\s+Build\//);
   179→  if (androidMatch) return { make: null, model: androidMatch[1]?.trim() ?? null };
   180→  return { make: null, model: null };
   181→}

   183→async function extractClientExif(
   184→  file: File,
   185→  deviceLocation: { latitude: number; longitude: number } | null,
   186→): Promise<ClientExifData | null> {
   187→  // Step 1: Try extracting EXIF from the raw file (works on Android, some iOS versions)
   188→  let lat: number | null = null;
   189→  let lng: number | null = null;
   190→  let capturedAt: string | null = null;
   191→  let deviceMake: string | null = null;
   192→  let deviceModel: string | null = null;
   193→
   194→  try {
   195→    const buffer = await file.arrayBuffer();
   196→    const [exifData, gpsData] = await Promise.all([
   197→      exifr.parse(buffer, { gps: true, exif: true, tiff: true, pick: [
   198→        'latitude', 'longitude', 'GPSLatitude', 'GPSLongitude',
   199→        'DateTimeOriginal', 'DateTimeDigitized', 'CreateDate', 'ModifyDate',
   200→        'Make', 'Model',
   201→      ] }).catch(() => null),
   202→      exifr.gps(buffer).catch(() => null),
   203→    ]);
   204→    const merged = { ...(exifData ?? {}), ...(gpsData ?? {}) } as Record<string, unknown>;
   205→    lat = typeof merged.latitude === 'number' ? merged.latitude : null;
   206→    lng = typeof merged.longitude === 'number' ? merged.longitude : null;
   207→    const rawDate = merged.DateTimeOriginal ?? merged.DateTimeDigitized ?? merged.CreateDate ?? merged.ModifyDate;
   208→    if (rawDate instanceof Date) {
   209→      capturedAt = rawDate.toISOString();
   210→    } else if (typeof rawDate === 'string' && rawDate.trim()) {
   211→      const m = rawDate.match(/^(\d{4}):(\d{2}):(\d{2})\s+(\d{2}):(\d{2}):(\d{2})/);
   212→      capturedAt = m ? `${m[1]}-${m[2]}-${m[3]}T${m[4]}:${m[5]}:${m[6]}` : rawDate;
   213→    }
   214→    deviceMake = typeof merged.Make === 'string' ? merged.Make : null;
   215→    deviceModel = typeof merged.Model === 'string' ? merged.Model : null;
   216→  } catch {
   217→    // EXIF parsing failed entirely -- fall through to device fallback
   218→  }
   219→
   220→  // Step 2: Fill gaps with browser APIs (critical for iOS which strips EXIF from <input capture>)
   221→  if (deviceLocation) {
   222→    if (lat == null) lat = deviceLocation.latitude;
   223→    if (lng == null) lng = deviceLocation.longitude;
   224→  }
   225→  if (!capturedAt) {
   226→    capturedAt = new Date().toISOString();
   227→  }
   228→  if (!deviceMake && !deviceModel) {
   229→    const parsed = parseDeviceFromUserAgent();
   230→    deviceMake = parsed.make;
   231→    deviceModel = parsed.model;
   232→  }
   233→
   234→  if (lat == null && lng == null && !capturedAt && !deviceMake && !deviceModel) return null;
   235→  return { latitude: lat, longitude: lng, capturedAt, deviceMake, deviceModel };
   236→}

  1447→  const verticalIcon = <VerticalIcon name={VERTICALS[vertical]?.icon ?? 'pill'} size={18} />;
  1448→
  1449→  if (submitted) {

  1489→          </button>

  1516→        {renderVerticalSelector()}

  1521→        {errorMessage && (
