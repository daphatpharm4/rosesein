    40→function dataUrlToBytes(dataUrl: string): Uint8Array | null {
    41→  const commaIndex = dataUrl.indexOf(',');
    42→  if (commaIndex === -1) return null;
    43→  const base64 = dataUrl.slice(commaIndex + 1);
    44→  if (!base64) return null;
    45→  const binary = atob(base64);
    46→  const bytes = new Uint8Array(binary.length);
    47→  for (let i = 0; i < binary.length; i += 1) {
    48→    bytes[i] = binary.charCodeAt(i);
    49→  }
    50→  return bytes;
    51→}

    53→function bytesToDataUrl(bytes: Uint8Array, mime: string): string {
    54→  let binary = '';
    55→  for (let i = 0; i < bytes.length; i += 1) {
    56→    binary += String.fromCharCode(bytes[i] ?? 0);
    57→  }
    58→  return `data:${mime};base64,${btoa(binary)}`;
    59→}

    61→function extractJpegExifSegment(bytes: Uint8Array): Uint8Array | null {
    62→  if (bytes.length < 4) return null;
    63→  if (bytes[0] !== 0xff || bytes[1] !== 0xd8) return null;
    64→
    65→  let offset = 2;
    66→  while (offset + 4 <= bytes.length) {
    67→    if (bytes[offset] !== 0xff) {
    68→      offset += 1;
    69→      continue;
    70→    }
    71→
    72→    const marker = bytes[offset + 1];
    73→    if (marker === 0xd8 || marker === 0x01 || (marker >= 0xd0 && marker <= 0xd7)) {
    74→      offset += 2;
    75→      continue;
    76→    }
    77→    if (marker === 0xda || marker === 0xd9) break;
    78→
    79→    const segmentLength = ((bytes[offset + 2] ?? 0) << 8) | (bytes[offset + 3] ?? 0);
    80→    if (segmentLength < 2) break;
    81→    const segmentEnd = offset + 2 + segmentLength;
    82→    if (segmentEnd > bytes.length) break;
    83→
    84→    if (marker === 0xe1 && segmentLength >= 8) {
    85→      const signature = String.fromCharCode(
    86→        bytes[offset + 4] ?? 0,
    87→        bytes[offset + 5] ?? 0,
    88→        bytes[offset + 6] ?? 0,
    89→        bytes[offset + 7] ?? 0,
    90→        bytes[offset + 8] ?? 0,
    91→        bytes[offset + 9] ?? 0,
    92→      );
    93→      if (signature === 'Exif\0\0') {
    94→        return bytes.slice(offset, segmentEnd);
    95→      }
    96→    }
    97→
    98→    offset = segmentEnd;
    99→  }
   100→
   101→  return null;
   102→}

   104→function injectExifIntoJpeg(compressedBytes: Uint8Array, exifSegment: Uint8Array): Uint8Array | null {
   105→  if (compressedBytes.length < 2) return null;
   106→  if (compressedBytes[0] !== 0xff || compressedBytes[1] !== 0xd8) return null;
   107→  if (!exifSegment.length) return null;
   108→
   109→  const merged = new Uint8Array(compressedBytes.length + exifSegment.length);
   110→  merged[0] = compressedBytes[0];
   111→  merged[1] = compressedBytes[1];
   112→  merged.set(exifSegment, 2);
   113→  merged.set(compressedBytes.slice(2), 2 + exifSegment.length);
   114→  return merged;
   115→}

   117→async function preserveJpegExif(originalFile: File, compressedDataUrl: string): Promise<string> {
   118→  if (!JPEG_MIME_TYPES.has((originalFile.type || '').toLowerCase())) return compressedDataUrl;
   119→
   120→  const originalBytes = new Uint8Array(await originalFile.arrayBuffer());
   121→  const exifSegment = extractJpegExifSegment(originalBytes);
   122→  if (!exifSegment) return compressedDataUrl;
   123→
   124→  const compressedBytes = dataUrlToBytes(compressedDataUrl);
   125→  if (!compressedBytes) return compressedDataUrl;
   126→  if (extractJpegExifSegment(compressedBytes)) return compressedDataUrl;
   127→
   128→  const merged = injectExifIntoJpeg(compressedBytes, exifSegment);
   129→  if (!merged) return compressedDataUrl;
   130→  return bytesToDataUrl(merged, 'image/jpeg');
   131→}

   133→function parseDeviceFromUserAgent(): { make: string | null; model: string | null } {
   134→  const ua = navigator.userAgent;
   135→  // iPhone detection
   136→  const iphoneMatch = ua.match(/iPhone/);
   137→  if (iphoneMatch) return { make: 'Apple', model: 'iPhone' };
   138→  // iPad detection
   139→  const ipadMatch = ua.match(/iPad/);
   140→  if (ipadMatch) return { make: 'Apple', model: 'iPad' };
   141→  // Android device detection
   142→  const androidMatch = ua.match(/;\s*([^;)]+)\s+Build\//);
   143→  if (androidMatch) return { make: null, model: androidMatch[1]?.trim() ?? null };
   144→  return { make: null, model: null };
   145→}

   147→async function extractClientExif(
   148→  file: File,
   149→  deviceLocation: { latitude: number; longitude: number } | null,
   150→): Promise<ClientExifData | null> {
   151→  // Step 1: Try extracting EXIF from the raw file (works on Android, some iOS versions)
   152→  let lat: number | null = null;
   153→  let lng: number | null = null;
   154→  let capturedAt: string | null = null;
   155→  let deviceMake: string | null = null;
   156→  let deviceModel: string | null = null;
   157→
   158→  try {
   159→    const buffer = await file.arrayBuffer();
   160→    const [exifData, gpsData] = await Promise.all([
   161→      exifr.parse(buffer, { gps: true, exif: true, tiff: true, pick: [
   162→        'latitude', 'longitude', 'GPSLatitude', 'GPSLongitude',
   163→        'DateTimeOriginal', 'DateTimeDigitized', 'CreateDate', 'ModifyDate',
   164→        'Make', 'Model',
   165→      ] }).catch(() => null),
   166→      exifr.gps(buffer).catch(() => null),
   167→    ]);
   168→    const merged = { ...(exifData ?? {}), ...(gpsData ?? {}) } as Record<string, unknown>;
   169→    lat = typeof merged.latitude === 'number' ? merged.latitude : null;
   170→    lng = typeof merged.longitude === 'number' ? merged.longitude : null;
   171→    const rawDate = merged.DateTimeOriginal ?? merged.DateTimeDigitized ?? merged.CreateDate ?? merged.ModifyDate;
   172→    if (rawDate instanceof Date) {
   173→      capturedAt = rawDate.toISOString();
   174→    } else if (typeof rawDate === 'string' && rawDate.trim()) {
   175→      const m = rawDate.match(/^(\d{4}):(\d{2}):(\d{2})\s+(\d{2}):(\d{2}):(\d{2})/);
   176→      capturedAt = m ? `${m[1]}-${m[2]}-${m[3]}T${m[4]}:${m[5]}:${m[6]}` : rawDate;
   177→    }
   178→    deviceMake = typeof merged.Make === 'string' ? merged.Make : null;
   179→    deviceModel = typeof merged.Model === 'string' ? merged.Model : null;
   180→  } catch {
   181→    // EXIF parsing failed entirely -- fall through to device fallback
   182→  }
   183→
   184→  // Step 2: Fill gaps with browser APIs (critical for iOS which strips EXIF from <input capture>)
   185→  if (lat == null && lng == null && deviceLocation) {
   186→    lat = deviceLocation.latitude;
   187→    lng = deviceLocation.longitude;
   188→  }
   189→  if (!capturedAt) {
   190→    capturedAt = new Date().toISOString();
   191→  }
   192→  if (!deviceMake && !deviceModel) {
   193→    const parsed = parseDeviceFromUserAgent();
   194→    deviceMake = parsed.make;
   195→    deviceModel = parsed.model;
   196→  }
   197→
   198→  if (lat == null && lng == null && !capturedAt && !deviceMake && !deviceModel) return null;
   199→  return { latitude: lat, longitude: lng, capturedAt, deviceMake, deviceModel };
   200→}
