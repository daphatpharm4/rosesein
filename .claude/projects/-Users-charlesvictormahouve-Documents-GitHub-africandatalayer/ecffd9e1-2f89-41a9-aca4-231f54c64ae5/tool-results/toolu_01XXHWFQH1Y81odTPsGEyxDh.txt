    41→function dataUrlToBytes(dataUrl: string): Uint8Array | null {
    42→  const commaIndex = dataUrl.indexOf(',');
    43→  if (commaIndex === -1) return null;
    44→  const base64 = dataUrl.slice(commaIndex + 1);
    45→  if (!base64) return null;
    46→  const binary = atob(base64);
    47→  const bytes = new Uint8Array(binary.length);
    48→  for (let i = 0; i < binary.length; i += 1) {
    49→    bytes[i] = binary.charCodeAt(i);
    50→  }
    51→  return bytes;
    52→}

    54→function bytesToDataUrl(bytes: Uint8Array, mime: string): string {
    55→  let binary = '';
    56→  for (let i = 0; i < bytes.length; i += 1) {
    57→    binary += String.fromCharCode(bytes[i] ?? 0);
    58→  }
    59→  return `data:${mime};base64,${btoa(binary)}`;
    60→}

    62→function extractJpegExifSegment(bytes: Uint8Array): Uint8Array | null {
    63→  if (bytes.length < 4) return null;
    64→  if (bytes[0] !== 0xff || bytes[1] !== 0xd8) return null;
    65→
    66→  let offset = 2;
    67→  while (offset + 4 <= bytes.length) {
    68→    if (bytes[offset] !== 0xff) {
    69→      offset += 1;
    70→      continue;
    71→    }
    72→
    73→    const marker = bytes[offset + 1];
    74→    if (marker === 0xd8 || marker === 0x01 || (marker >= 0xd0 && marker <= 0xd7)) {
    75→      offset += 2;
    76→      continue;
    77→    }
    78→    if (marker === 0xda || marker === 0xd9) break;
    79→
    80→    const segmentLength = ((bytes[offset + 2] ?? 0) << 8) | (bytes[offset + 3] ?? 0);
    81→    if (segmentLength < 2) break;
    82→    const segmentEnd = offset + 2 + segmentLength;
    83→    if (segmentEnd > bytes.length) break;
    84→
    85→    if (marker === 0xe1 && segmentLength >= 8) {
    86→      const signature = String.fromCharCode(
    87→        bytes[offset + 4] ?? 0,
    88→        bytes[offset + 5] ?? 0,
    89→        bytes[offset + 6] ?? 0,
    90→        bytes[offset + 7] ?? 0,
    91→        bytes[offset + 8] ?? 0,
    92→        bytes[offset + 9] ?? 0,
    93→      );
    94→      if (signature === 'Exif\0\0') {
    95→        return bytes.slice(offset, segmentEnd);
    96→      }
    97→    }
    98→
    99→    offset = segmentEnd;
   100→  }
   101→
   102→  return null;
   103→}

   105→function injectExifIntoJpeg(compressedBytes: Uint8Array, exifSegment: Uint8Array): Uint8Array | null {
   106→  if (compressedBytes.length < 2) return null;
   107→  if (compressedBytes[0] !== 0xff || compressedBytes[1] !== 0xd8) return null;
   108→  if (!exifSegment.length) return null;
   109→
   110→  const merged = new Uint8Array(compressedBytes.length + exifSegment.length);
   111→  merged[0] = compressedBytes[0];
   112→  merged[1] = compressedBytes[1];
   113→  merged.set(exifSegment, 2);
   114→  merged.set(compressedBytes.slice(2), 2 + exifSegment.length);
   115→  return merged;
   116→}

   118→async function preserveJpegExif(originalFile: File, compressedDataUrl: string): Promise<string> {
   119→  if (!JPEG_MIME_TYPES.has((originalFile.type || '').toLowerCase())) return compressedDataUrl;
   120→
   121→  const originalBytes = new Uint8Array(await originalFile.arrayBuffer());
   122→  const exifSegment = extractJpegExifSegment(originalBytes);
   123→  if (!exifSegment) return compressedDataUrl;
   124→
   125→  const compressedBytes = dataUrlToBytes(compressedDataUrl);
   126→  if (!compressedBytes) return compressedDataUrl;
   127→  if (extractJpegExifSegment(compressedBytes)) return compressedDataUrl;
   128→
   129→  const merged = injectExifIntoJpeg(compressedBytes, exifSegment);
   130→  if (!merged) return compressedDataUrl;
   131→  return bytesToDataUrl(merged, 'image/jpeg');
   132→}

   134→function parseDeviceFromUserAgent(): { make: string | null; model: string | null } {
   135→  const ua = navigator.userAgent;
   136→  // iPhone detection
   137→  const iphoneMatch = ua.match(/iPhone/);
   138→  if (iphoneMatch) return { make: 'Apple', model: 'iPhone' };
   139→  // iPad detection
   140→  const ipadMatch = ua.match(/iPad/);
   141→  if (ipadMatch) return { make: 'Apple', model: 'iPad' };
   142→  // Android device detection
   143→  const androidMatch = ua.match(/;\s*([^;)]+)\s+Build\//);
   144→  if (androidMatch) return { make: null, model: androidMatch[1]?.trim() ?? null };
   145→  return { make: null, model: null };
   146→}

   148→async function extractClientExif(
   149→  file: File,
   150→  deviceLocation: { latitude: number; longitude: number } | null,
   151→): Promise<ClientExifData | null> {
   152→  // Step 1: Try extracting EXIF from the raw file (works on Android, some iOS versions)
   153→  let lat: number | null = null;
   154→  let lng: number | null = null;
   155→  let capturedAt: string | null = null;
   156→  let deviceMake: string | null = null;
   157→  let deviceModel: string | null = null;
   158→
   159→  try {
   160→    const buffer = await file.arrayBuffer();
   161→    const [exifData, gpsData] = await Promise.all([
   162→      exifr.parse(buffer, { gps: true, exif: true, tiff: true, pick: [
   163→        'latitude', 'longitude', 'GPSLatitude', 'GPSLongitude',
   164→        'DateTimeOriginal', 'DateTimeDigitized', 'CreateDate', 'ModifyDate',
   165→        'Make', 'Model',
   166→      ] }).catch(() => null),
   167→      exifr.gps(buffer).catch(() => null),
   168→    ]);
   169→    const merged = { ...(exifData ?? {}), ...(gpsData ?? {}) } as Record<string, unknown>;
   170→    lat = typeof merged.latitude === 'number' ? merged.latitude : null;
   171→    lng = typeof merged.longitude === 'number' ? merged.longitude : null;
   172→    const rawDate = merged.DateTimeOriginal ?? merged.DateTimeDigitized ?? merged.CreateDate ?? merged.ModifyDate;
   173→    if (rawDate instanceof Date) {
   174→      capturedAt = rawDate.toISOString();
   175→    } else if (typeof rawDate === 'string' && rawDate.trim()) {
   176→      const m = rawDate.match(/^(\d{4}):(\d{2}):(\d{2})\s+(\d{2}):(\d{2}):(\d{2})/);
   177→      capturedAt = m ? `${m[1]}-${m[2]}-${m[3]}T${m[4]}:${m[5]}:${m[6]}` : rawDate;
   178→    }
   179→    deviceMake = typeof merged.Make === 'string' ? merged.Make : null;
   180→    deviceModel = typeof merged.Model === 'string' ? merged.Model : null;
   181→  } catch {
   182→    // EXIF parsing failed entirely -- fall through to device fallback
   183→  }
   184→
   185→  // Step 2: Fill gaps with browser APIs (critical for iOS which strips EXIF from <input capture>)
   186→  if (lat == null && lng == null && deviceLocation) {
   187→    lat = deviceLocation.latitude;
   188→    lng = deviceLocation.longitude;
   189→  }
   190→  if (!capturedAt) {
   191→    capturedAt = new Date().toISOString();
   192→  }
   193→  if (!deviceMake && !deviceModel) {
   194→    const parsed = parseDeviceFromUserAgent();
   195→    deviceMake = parsed.make;
   196→    deviceModel = parsed.model;
   197→  }
   198→
   199→  if (lat == null && lng == null && !capturedAt && !deviceMake && !deviceModel) return null;
   200→  return { latitude: lat, longitude: lng, capturedAt, deviceMake, deviceModel };
   201→}
