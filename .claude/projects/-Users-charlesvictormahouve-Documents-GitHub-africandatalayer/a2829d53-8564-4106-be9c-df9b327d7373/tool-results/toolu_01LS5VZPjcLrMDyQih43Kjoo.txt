   249→function normalizeIp(raw: string | null): string | null {
   250→  if (!raw) return null;
   251→  const first = raw.split(",")[0]?.trim();
   252→  if (!first) return null;
   253→  const cleaned = first.replace(/^\[|\]$/g, "");
   254→  if (cleaned.startsWith("::ffff:")) return cleaned.slice(7);
   255→  if (/^\d+\.\d+\.\d+\.\d+:\d+$/.test(cleaned)) return cleaned.split(":")[0] ?? null;
   256→  return cleaned;
   257→}

   259→function isPrivateIp(ip: string): boolean {
   260→  if (ip === "127.0.0.1" || ip === "::1") return true;
   261→  if (ip.startsWith("10.")) return true;
   262→  if (ip.startsWith("192.168.")) return true;
   263→  if (ip.startsWith("172.")) {
   264→    const second = Number(ip.split(".")[1]);
   265→    if (second >= 16 && second <= 31) return true;
   266→  }
   267→  if (ip.startsWith("fc") || ip.startsWith("fd")) return true;
   268→  if (ip.startsWith("fe80:")) return true;
   269→  return false;
   270→}

   317→async function fetchIpInfo(ip: string): Promise<IpLookupResult> {
   318→  const target = `https://ipapi.co/${ip}/json/`;
   319→  const controller = new AbortController();
   320→  const timeout = setTimeout(() => controller.abort(), 3000);
   321→  try {
   322→    const res = await fetch(target, { signal: controller.signal });
   323→    if (!res.ok) {
   324→      return {
   325→        ip,
   326→        location: null,
   327→        asn: null,
   328→        org: null,
   329→        isDatacenter: false,
   330→        isVpn: false,
   331→        isTor: false,
   332→      };
   333→    }
   334→    const data = (await res.json()) as IpApiResponse;
   335→    const latitude = Number(data?.latitude);
   336→    const longitude = Number(data?.longitude);
   337→    const org = toLowerText(data?.org) || null;
   338→    const asn = typeof data?.asn === "string" && data.asn.trim() ? data.asn.trim().toLowerCase() : null;
   339→    const classification = classifyIpOrg(org ?? "");
   340→    return {
   341→      ip,
   342→      location: Number.isFinite(latitude) && Number.isFinite(longitude) ? { latitude, longitude } : null,
   343→      asn,
   344→      org,
   345→      isDatacenter: classification.isDatacenter,
   346→      isVpn: classification.isVpn,
   347→      isTor: classification.isTor,
   348→    };
   349→  } catch {
   350→    return {
   351→      ip,
   352→      location: null,
   353→      asn: null,
   354→      org: null,
   355→      isDatacenter: false,
   356→      isVpn: false,
   357→      isTor: false,
   358→    };
   359→  } finally {
   360→    clearTimeout(timeout);
   361→  }
   362→}

   454→async function buildFallbackFraudCheck(
   455→  event: PointEvent,
   456→  existingFraudCheck: SubmissionFraudCheck | null,
   457→): Promise<SubmissionFraudCheck | null> {
   458→  const primaryUrl = typeof event.photoUrl === "string" ? event.photoUrl.trim() : "";
   459→  const secondaryUrl = getSecondaryPhotoUrl(event);
   460→  if (!primaryUrl && !secondaryUrl) return existingFraudCheck;
   461→
   462→  const needsPrimaryRecovery = Boolean(primaryUrl) && isPhotoMetadataEffectivelyEmpty(existingFraudCheck?.primaryPhoto);
   463→  const needsSecondaryRecovery = Boolean(secondaryUrl) && isPhotoMetadataEffectivelyEmpty(existingFraudCheck?.secondaryPhoto);
   464→  const primaryExtracted = needsPrimaryRecovery ? await extractPhotoMetadataFromUrl(primaryUrl) : null;
   465→  const secondaryExtracted = needsSecondaryRecovery ? await extractPhotoMetadataFromUrl(secondaryUrl) : null;
   466→  const recoveredPrimaryPhoto = buildPhotoFraudMetadata({
   467→    extracted: primaryExtracted,
   468→    submissionLocation: event.location,
   469→    ipLocation: null,
   470→    submissionMatchThresholdKm: SUBMISSION_PHOTO_MATCH_KM,
   471→    ipMatchThresholdKm: IP_PHOTO_MATCH_KM,
   472→  });
   473→  const recoveredSecondaryPhoto = buildPhotoFraudMetadata({
   474→    extracted: secondaryExtracted,
   475→    submissionLocation: event.location,
   476→    ipLocation: null,
   477→    submissionMatchThresholdKm: SUBMISSION_PHOTO_MATCH_KM,
   478→    ipMatchThresholdKm: IP_PHOTO_MATCH_KM,
   479→  });
   480→  const primaryPhoto = mergePhotoFraudMetadata(existingFraudCheck?.primaryPhoto ?? null, recoveredPrimaryPhoto, Boolean(primaryUrl));
   481→  const secondaryPhoto = mergePhotoFraudMetadata(
   482→    existingFraudCheck?.secondaryPhoto ?? null,
   483→    recoveredSecondaryPhoto,
   484→    Boolean(secondaryUrl),
   485→  );
   486→  if (!primaryPhoto && !secondaryPhoto) return null;
   487→
   488→  return buildSubmissionFraudCheck({
   489→    submissionLocation: existingFraudCheck?.submissionLocation ?? event.location,
   490→    effectiveLocation: existingFraudCheck?.effectiveLocation ?? event.location,
   491→    ipLocation: existingFraudCheck?.ipLocation ?? null,
   492→    primaryPhoto,
   493→    secondaryPhoto,
   494→    submissionMatchThresholdKm: existingFraudCheck?.submissionMatchThresholdKm ?? SUBMISSION_PHOTO_MATCH_KM,
   495→    ipMatchThresholdKm: existingFraudCheck?.ipMatchThresholdKm ?? IP_PHOTO_MATCH_KM,
   496→  });
   497→}

   499→async function buildAdminSubmissionEvents(events: PointEvent[]): Promise<AdminSubmissionEvent[]> {
   500→  const sortedEvents = [...events].sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
   501→  const userIds = Array.from(
   502→    new Set(
   503→      sortedEvents
   504→        .map((event) => normalizeActorId(event.userId))
   505→        .filter((value) => value.length > 0),
   506→    ),
   507→  );
   508→
   509→  const profileEntries = await Promise.all(
   510→    userIds.map(async (userId) => [userId, await getUserProfile(userId)] as const),
   511→  );
   512→  const profileByUserId = new Map(profileEntries);
   513→  const output: AdminSubmissionEvent[] = [];
   514→  let fallbackLookups = 0;
   515→
   516→  for (const event of sortedEvents) {
   517→    const normalizedUserId = normalizeActorId(event.userId);
   518→    const profile = profileByUserId.get(normalizedUserId) ?? null;
   519→    const emailFromProfile = typeof profile?.email === "string" ? profile.email.trim().toLowerCase() : "";
   520→    const email = emailFromProfile || (normalizedUserId.includes("@") ? normalizedUserId : null);
   521→    const name = getUserDisplayName(normalizedUserId, profile?.name ?? null, email);
   522→    let fraudCheck = getEventFraudCheck(event);
   523→    if (shouldRunFallbackFraudCheck(event, fraudCheck) && fallbackLookups < MAX_FORENSICS_FALLBACK_LOOKUPS) {
   524→      fallbackLookups += 1;
   525→      try {
   526→        fraudCheck = await buildFallbackFraudCheck(event, fraudCheck);
   527→      } catch {
   528→        fraudCheck = fraudCheck ?? null;
   529→      }
   530→    }
   531→
   532→    output.push({
   533→      event,
   534→      user: {
   535→        id: normalizedUserId || event.userId,
   536→        name,
   537→        email,
   538→        trustScore: profile?.trustScore ?? 50,
   539→        trustTier: profile?.trustTier ?? "standard",
   540→        suspendedUntil: profile?.suspendedUntil ?? null,
   541→      },
   542→      fraudCheck,
   543→    });
   544→  }
   545→
   546→  return output;
   547→}

   549→type SchemaGuardView = {
   550→  ok: boolean | null;
   551→  expected: string[];
   552→  actual: string[];
   553→  missing: string[];
   554→  extra: string[];
   555→  reason?: string;
   556→};
   557→
   558→async function buildSchemaGuardView(): Promise<SchemaGuardView> {
   559→  const fallback: SchemaGuardView = {
   560→    ok: null,
   561→    expected: EXPECTED_SUBMISSION_CATEGORIES,
   562→    actual: [],
   563→    missing: EXPECTED_SUBMISSION_CATEGORIES,
   564→    extra: [],
   565→  };
   566→
   567→  try {
   568→    const result = await query<{ constraint_definition: string | null }>(
   569→      `
   570→        SELECT pg_get_constraintdef(c.oid) AS constraint_definition
   571→        FROM pg_constraint c
   572→        JOIN pg_class rel ON rel.oid = c.conrelid
   573→        JOIN pg_namespace ns ON ns.oid = rel.relnamespace
   574→        WHERE ns.nspname = 'public'
   575→          AND rel.relname = 'point_events'
   576→          AND c.conname = 'point_events_category_check'
   577→        LIMIT 1
   578→      `,
   579→    );
   580→    const definition = result.rows[0]?.constraint_definition ?? null;
   581→    if (!definition) {
   582→      return { ...fallback, reason: "constraint_not_found" };
   583→    }
   584→    const actual = parseConstraintCategories(definition);
   585→    const diff = diffCategorySets(EXPECTED_SUBMISSION_CATEGORIES, actual);
   586→    return {
   587→      ok: diff.ok,
   588→      expected: diff.expected,
   589→      actual: diff.actual,
   590→      missing: diff.missing,
   591→      extra: diff.extra,
   592→    };
   593→  } catch (error) {
   594→    if (isStorageUnavailableError(error)) {
   595→      return { ...fallback, reason: "storage_unavailable" };
   596→    }
   597→    return { ...fallback, reason: "query_failed" };
   598→  }
   599→}

   601→export async function GET(request: Request): Promise<Response> {
   602→  const auth = await requireUser(request);
   603→  const authContext = toSubmissionAuthContext(auth);
   604→  const url = new URL(request.url);
   605→  const view = url.searchParams.get("view");
   606→  const lat = url.searchParams.get("lat");
   607→  const lng = url.searchParams.get("lng");
   608→  const radius = url.searchParams.get("radius");
   609→  const requestedScope = normalizeMapScope(url.searchParams.get("scope"));
   610→  const canUseExpandedScope = requestedScope !== "bonamoussadi";
   611→  if (canUseExpandedScope && !authContext) return errorResponse("Unauthorized", 401);
   612→  if (canUseExpandedScope && !authContext.isAdmin) return errorResponse("Forbidden", 403);
   613→  const effectiveScope = canUseExpandedScope ? requestedScope : "bonamoussadi";
   614→  const canUsePublicCache = !authContext && !canUseExpandedScope;
   615→
   616→  try {
   617→    if (view === "schema_guard") {
   618→      const adminAccess = resolveAdminViewAccess(authContext);
   619→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   620→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   621→      return jsonResponse(await buildSchemaGuardView(), { status: 200 });
   622→    }
   623→
   624→    const allEvents = await buildReadableEvents();
   625→    const scopedEvents = allEvents.filter((event) => {
   626→      if (effectiveScope === "global") return true;
   627→      if (effectiveScope === "cameroon") return isWithinCameroon(event.location);
   628→      return isWithinBonamoussadi(event.location);
   629→    });
   630→
   631→    if (view === "events") {
   632→      if (!authContext) {
   633→        return jsonResponse(redactEventUserIds(scopedEvents), {
   634→          status: 200,
   635→          headers: {
   636→            "cache-control": PUBLIC_READ_CACHE_CONTROL,
   637→          },
   638→        });
   639→      }
   640→      const responseEvents = filterEventsForViewer(scopedEvents, authContext);
   641→      return jsonResponse(responseEvents, { status: 200 });
   642→    }
   643→
   644→    if (view === "admin_events") {
   645→      const adminAccess = resolveAdminViewAccess(authContext);
   646→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   647→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   648→      const adminEvents = await buildAdminSubmissionEvents(scopedEvents);
   649→      return jsonResponse(adminEvents, { status: 200 });
   650→    }
   651→
   652→    if (view === "review_queue") {
   653→      const adminAccess = resolveAdminViewAccess(authContext);
   654→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   655→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   656→      const pending = scopedEvents.filter((event) => isPendingReview(event));
   657→      const adminEvents = await buildAdminSubmissionEvents(pending);
   658→      return jsonResponse(adminEvents, { status: 200 });
   659→    }
   660→
   661→    let projected = projectPointsFromEvents(scopedEvents);
   662→
   663→    if (view === "dedup_candidates") {
   664→      if (!authContext) return errorResponse("Unauthorized", 401);
   665→      const category = normalizeCategory(url.searchParams.get("category") ?? undefined);
   666→      if (!category) return errorResponse("Invalid category", 400);
   667→      const latitude = Number(url.searchParams.get("lat"));
   668→      const longitude = Number(url.searchParams.get("lng"));
   669→      if (!Number.isFinite(latitude) || !Number.isFinite(longitude)) {
   670→        return errorResponse("lat and lng are required", 400);
   671→      }
   672→      const name = trimString(url.searchParams.get("name"), 160);
   673→      const probeDetails: SubmissionDetails = {};
   674→      if (name) {
   675→        probeDetails.name = name;
   676→        probeDetails.siteName = name;
   677→        probeDetails.roadName = name;
   678→      }
   679→      const result = buildDedupCandidates(
   680→        category,
   681→        { latitude, longitude },
   682→        probeDetails,
   683→        projected,
   684→      );
   685→      return jsonResponse(result);
   686→    }
   687→
   688→    if (lat && lng && radius) {
   689→      const latitude = Number(lat);
   690→      const longitude = Number(lng);
   691→      const radiusKm = Number(radius);
   692→      if (Number.isFinite(latitude) && Number.isFinite(longitude) && Number.isFinite(radiusKm)) {
   693→        projected = projected.filter((point) => haversineKm(point.location, { latitude, longitude }) <= radiusKm);
   694→      }
   695→    }
   696→
   697→    return jsonResponse(projected, {
   698→      status: 200,
   699→      headers: canUsePublicCache
   700→        ? {
   701→            "cache-control": PUBLIC_READ_CACHE_CONTROL,
   702→          }
   703→        : undefined,
   704→    });
   705→  } catch (error) {
   706→    if (isStorageUnavailableError(error)) {
   707→      return errorResponse("Storage service temporarily unavailable", 503, { code: "storage_unavailable" });
   708→    }
   709→    throw error;
   710→  }
   711→}

  1008→        ipHash,
  1009→        ipReputation: ipInfo
  1010→          ? {
  1011→              asn: ipInfo.asn,
  1012→              org: ipInfo.org,

  1014→              isVpn: ipInfo.isVpn,

  1020→    } catch (riskError) {

  1122→    } catch (error) {

  1124→      const lower = raw.toLowerCase();

  1126→      return errorResponse("Unable to store photo", 500);

  1158→      try {

  1160→          `${eventId}-second`,

  1164→        );
