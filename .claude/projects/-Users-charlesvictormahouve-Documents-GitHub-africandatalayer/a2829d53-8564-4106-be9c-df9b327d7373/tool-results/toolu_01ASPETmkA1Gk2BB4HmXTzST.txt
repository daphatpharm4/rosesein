   220→function groupEventsByPoint(items: AdminSubmissionEvent[], language: 'en' | 'fr'): GroupedPoint[] {
   221→  const groups = new Map<string, AdminSubmissionEvent[]>();
   222→  for (const item of items) {
   223→    const pid = item.event.pointId;
   224→    const existing = groups.get(pid);
   225→    if (existing) {
   226→      existing.push(item);
   227→    } else {
   228→      groups.set(pid, [item]);
   229→    }
   230→  }
   231→  const result: GroupedPoint[] = [];
   232→  for (const [pointId, events] of groups) {
   233→    const sorted = [...events].sort((a, b) => new Date(a.event.createdAt).getTime() - new Date(b.event.createdAt).getTime());
   234→    const latestEvent = sorted[sorted.length - 1]!;
   235→    const createdEvent = sorted.find((e) => e.event.eventType === 'CREATE_EVENT') ?? null;
   236→    const enrichEvents = sorted.filter((e) => e.event.eventType === 'ENRICH_EVENT');
   237→    const allPhotos: GroupedPoint['allPhotos'] = [];
   238→    for (const ev of sorted) {
   239→      const photoUrl = ev.event.photoUrl;
   240→      if (photoUrl && typeof photoUrl === 'string' && photoUrl.trim()) {
   241→        allPhotos.push({
   242→          url: photoUrl,
   243→          eventType: ev.event.eventType,
   244→          createdAt: ev.event.createdAt,
   245→          metadata: ev.fraudCheck?.primaryPhoto ?? null,
   246→        });
   247→      }
   248→      const details = ev.event.details as SubmissionDetails;
   249→      const secondUrl = typeof details.secondPhotoUrl === 'string' && details.secondPhotoUrl.trim() ? details.secondPhotoUrl : null;
   250→      if (secondUrl) {
   251→        allPhotos.push({
   252→          url: secondUrl,
   253→          eventType: ev.event.eventType + ' (secondary)',
   254→          createdAt: ev.event.createdAt,
   255→          metadata: ev.fraudCheck?.secondaryPhoto ?? null,
   256→        });
   257→      }
   258→    }
   259→    result.push({
   260→      pointId,
   261→      events: sorted,
   262→      category: latestEvent.event.category,
   263→      siteName: getSiteName(createdEvent ?? latestEvent, language),
   264→      latestEvent,
   265→      createdEvent,
   266→      enrichEvents,
   267→      allPhotos,
   268→    });
   269→  }
   270→  return result.sort((a, b) => new Date(b.latestEvent.event.createdAt).getTime() - new Date(a.latestEvent.event.createdAt).getTime());
   271→}
