   249→function normalizeIp(raw: string | null): string | null {
   250→  if (!raw) return null;
   251→  const first = raw.split(",")[0]?.trim();
   252→  if (!first) return null;
   253→  const cleaned = first.replace(/^\[|\]$/g, "");
   254→  if (cleaned.startsWith("::ffff:")) return cleaned.slice(7);
   255→  if (/^\d+\.\d+\.\d+\.\d+:\d+$/.test(cleaned)) return cleaned.split(":")[0] ?? null;
   256→  return cleaned;
   257→}

   259→function isPrivateIp(ip: string): boolean {
   260→  if (ip === "127.0.0.1" || ip === "::1") return true;
   261→  if (ip.startsWith("10.")) return true;
   262→  if (ip.startsWith("192.168.")) return true;
   263→  if (ip.startsWith("172.")) {
   264→    const second = Number(ip.split(".")[1]);
   265→    if (second >= 16 && second <= 31) return true;
   266→  }
   267→  if (ip.startsWith("fc") || ip.startsWith("fd")) return true;
   268→  if (ip.startsWith("fe80:")) return true;
   269→  return false;
   270→}

   317→async function fetchIpInfo(ip: string): Promise<IpLookupResult> {
   318→  const target = `https://ipapi.co/${ip}/json/`;
   319→  const controller = new AbortController();
   320→  const timeout = setTimeout(() => controller.abort(), 3000);
   321→  try {
   322→    const res = await fetch(target, { signal: controller.signal });
   323→    if (!res.ok) {
   324→      return {
   325→        ip,
   326→        location: null,
   327→        asn: null,
   328→        org: null,
   329→        isDatacenter: false,
   330→        isVpn: false,
   331→        isTor: false,
   332→      };
   333→    }
   334→    const data = (await res.json()) as IpApiResponse;
   335→    const latitude = Number(data?.latitude);
   336→    const longitude = Number(data?.longitude);
   337→    const org = toLowerText(data?.org) || null;
   338→    const asn = typeof data?.asn === "string" && data.asn.trim() ? data.asn.trim().toLowerCase() : null;
   339→    const classification = classifyIpOrg(org ?? "");
   340→    return {
   341→      ip,
   342→      location: Number.isFinite(latitude) && Number.isFinite(longitude) ? { latitude, longitude } : null,
   343→      asn,
   344→      org,
   345→      isDatacenter: classification.isDatacenter,
   346→      isVpn: classification.isVpn,
   347→      isTor: classification.isTor,
   348→    };
   349→  } catch {
   350→    return {
   351→      ip,
   352→      location: null,
   353→      asn: null,
   354→      org: null,
   355→      isDatacenter: false,
   356→      isVpn: false,
   357→      isTor: false,
   358→    };
   359→  } finally {
   360→    clearTimeout(timeout);
   361→  }
   362→}

   454→async function buildFallbackFraudCheck(
   455→  event: PointEvent,
   456→  existingFraudCheck: SubmissionFraudCheck | null,
   457→): Promise<SubmissionFraudCheck | null> {
   458→  const primaryUrl = typeof event.photoUrl === "string" ? event.photoUrl.trim() : "";
   459→  const secondaryUrl = getSecondaryPhotoUrl(event);
   460→  if (!primaryUrl && !secondaryUrl) return existingFraudCheck;
   461→
   462→  const needsPrimaryRecovery = Boolean(primaryUrl) && isPhotoMetadataEffectivelyEmpty(existingFraudCheck?.primaryPhoto);
   463→  const needsSecondaryRecovery = Boolean(secondaryUrl) && isPhotoMetadataEffectivelyEmpty(existingFraudCheck?.secondaryPhoto);
   464→  const primaryExtracted = needsPrimaryRecovery ? await extractPhotoMetadataFromUrl(primaryUrl) : null;
   465→  const secondaryExtracted = needsSecondaryRecovery ? await extractPhotoMetadataFromUrl(secondaryUrl) : null;
   466→  const recoveredPrimaryPhoto = buildPhotoFraudMetadata({
   467→    extracted: primaryExtracted,
   468→    submissionLocation: event.location,
   469→    ipLocation: null,
   470→    submissionMatchThresholdKm: SUBMISSION_PHOTO_MATCH_KM,
   471→    ipMatchThresholdKm: IP_PHOTO_MATCH_KM,
   472→  });
   473→  const recoveredSecondaryPhoto = buildPhotoFraudMetadata({
   474→    extracted: secondaryExtracted,
   475→    submissionLocation: event.location,
   476→    ipLocation: null,
   477→    submissionMatchThresholdKm: SUBMISSION_PHOTO_MATCH_KM,
   478→    ipMatchThresholdKm: IP_PHOTO_MATCH_KM,
   479→  });
   480→  const primaryPhoto = mergePhotoFraudMetadata(existingFraudCheck?.primaryPhoto ?? null, recoveredPrimaryPhoto, Boolean(primaryUrl));
   481→  const secondaryPhoto = mergePhotoFraudMetadata(
   482→    existingFraudCheck?.secondaryPhoto ?? null,
   483→    recoveredSecondaryPhoto,
   484→    Boolean(secondaryUrl),
   485→  );
   486→  if (!primaryPhoto && !secondaryPhoto) return null;
   487→
   488→  return buildSubmissionFraudCheck({
   489→    submissionLocation: existingFraudCheck?.submissionLocation ?? event.location,
   490→    effectiveLocation: existingFraudCheck?.effectiveLocation ?? event.location,
   491→    ipLocation: existingFraudCheck?.ipLocation ?? null,
   492→    primaryPhoto,
   493→    secondaryPhoto,
   494→    submissionMatchThresholdKm: existingFraudCheck?.submissionMatchThresholdKm ?? SUBMISSION_PHOTO_MATCH_KM,
   495→    ipMatchThresholdKm: existingFraudCheck?.ipMatchThresholdKm ?? IP_PHOTO_MATCH_KM,
   496→  });
   497→}

   499→async function buildAdminSubmissionEvents(events: PointEvent[]): Promise<AdminSubmissionEvent[]> {
   500→  const sortedEvents = [...events].sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
   501→  const userIds = Array.from(
   502→    new Set(
   503→      sortedEvents
   504→        .map((event) => normalizeActorId(event.userId))
   505→        .filter((value) => value.length > 0),
   506→    ),
   507→  );
   508→
   509→  const profileEntries = await Promise.all(
   510→    userIds.map(async (userId) => [userId, await getUserProfile(userId)] as const),
   511→  );
   512→  const profileByUserId = new Map(profileEntries);
   513→  const output: AdminSubmissionEvent[] = [];
   514→  let fallbackLookups = 0;
   515→
   516→  for (const event of sortedEvents) {
   517→    const normalizedUserId = normalizeActorId(event.userId);
   518→    const profile = profileByUserId.get(normalizedUserId) ?? null;
   519→    const emailFromProfile = typeof profile?.email === "string" ? profile.email.trim().toLowerCase() : "";
   520→    const email = emailFromProfile || (normalizedUserId.includes("@") ? normalizedUserId : null);
   521→    const name = getUserDisplayName(normalizedUserId, profile?.name ?? null, email);
   522→    let fraudCheck = getEventFraudCheck(event);
   523→    if (shouldRunFallbackFraudCheck(event, fraudCheck) && fallbackLookups < MAX_FORENSICS_FALLBACK_LOOKUPS) {
   524→      fallbackLookups += 1;
   525→      try {
   526→        fraudCheck = await buildFallbackFraudCheck(event, fraudCheck);
   527→      } catch {
   528→        fraudCheck = fraudCheck ?? null;
   529→      }
   530→    }
   531→
   532→    output.push({
   533→      event,
   534→      user: {
   535→        id: normalizedUserId || event.userId,
   536→        name,
   537→        email,
   538→        avatarPreset: profile?.avatarPreset ?? undefined,
   539→        trustScore: profile?.trustScore ?? 50,
   540→        trustTier: profile?.trustTier ?? "standard",
   541→        suspendedUntil: profile?.suspendedUntil ?? null,
   542→      },
   543→      fraudCheck,
   544→    });
   545→  }
   546→
   547→  return output;
   548→}

   550→type SchemaGuardView = {
   551→  ok: boolean | null;
   552→  expected: string[];
   553→  actual: string[];
   554→  missing: string[];
   555→  extra: string[];
   556→  reason?: string;
   557→};

   559→async function buildSchemaGuardView(): Promise<SchemaGuardView> {
   560→  const fallback: SchemaGuardView = {
   561→    ok: null,
   562→    expected: EXPECTED_SUBMISSION_CATEGORIES,
   563→    actual: [],
   564→    missing: EXPECTED_SUBMISSION_CATEGORIES,
   565→    extra: [],
   566→  };
   567→
   568→  try {
   569→    const result = await query<{ constraint_definition: string | null }>(
   570→      `
   571→        SELECT pg_get_constraintdef(c.oid) AS constraint_definition
   572→        FROM pg_constraint c
   573→        JOIN pg_class rel ON rel.oid = c.conrelid
   574→        JOIN pg_namespace ns ON ns.oid = rel.relnamespace
   575→        WHERE ns.nspname = 'public'
   576→          AND rel.relname = 'point_events'
   577→          AND c.conname = 'point_events_category_check'
   578→        LIMIT 1
   579→      `,
   580→    );
   581→    const definition = result.rows[0]?.constraint_definition ?? null;
   582→    if (!definition) {
   583→      return { ...fallback, reason: "constraint_not_found" };
   584→    }
   585→    const actual = parseConstraintCategories(definition);
   586→    const diff = diffCategorySets(EXPECTED_SUBMISSION_CATEGORIES, actual);
   587→    return {
   588→      ok: diff.ok,
   589→      expected: diff.expected,
   590→      actual: diff.actual,
   591→      missing: diff.missing,
   592→      extra: diff.extra,
   593→    };
   594→  } catch (error) {
   595→    if (isStorageUnavailableError(error)) {
   596→      return { ...fallback, reason: "storage_unavailable" };
   597→    }
   598→    return { ...fallback, reason: "query_failed" };
   599→  }
   600→}

   602→export async function GET(request: Request): Promise<Response> {
   603→  const auth = await requireUser(request);
   604→  const authContext = toSubmissionAuthContext(auth);
   605→  const url = new URL(request.url);
   606→  const view = url.searchParams.get("view");
   607→  const lat = url.searchParams.get("lat");
   608→  const lng = url.searchParams.get("lng");
   609→  const radius = url.searchParams.get("radius");
   610→  const requestedScope = normalizeMapScope(url.searchParams.get("scope"));
   611→  const canUseExpandedScope = requestedScope !== "bonamoussadi";
   612→  if (canUseExpandedScope && !authContext) return errorResponse("Unauthorized", 401);
   613→  if (canUseExpandedScope && !authContext.isAdmin) return errorResponse("Forbidden", 403);
   614→  const effectiveScope = canUseExpandedScope ? requestedScope : "bonamoussadi";
   615→  const canUsePublicCache = !authContext && !canUseExpandedScope;
   616→
   617→  try {
   618→    if (view === "schema_guard") {
   619→      const adminAccess = resolveAdminViewAccess(authContext);
   620→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   621→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   622→      return jsonResponse(await buildSchemaGuardView(), { status: 200 });
   623→    }
   624→
   625→    const allEvents = await buildReadableEvents();
   626→    const scopedEvents = allEvents.filter((event) => {
   627→      if (effectiveScope === "global") return true;
   628→      if (effectiveScope === "cameroon") return isWithinCameroon(event.location);
   629→      return isWithinBonamoussadi(event.location);
   630→    });
   631→
   632→    if (view === "events") {
   633→      if (!authContext) {
   634→        return jsonResponse(redactEventUserIds(scopedEvents), {
   635→          status: 200,
   636→          headers: {
   637→            "cache-control": PUBLIC_READ_CACHE_CONTROL,
   638→          },
   639→        });
   640→      }
   641→      const responseEvents = filterEventsForViewer(scopedEvents, authContext);
   642→      return jsonResponse(responseEvents, { status: 200 });
   643→    }
   644→
   645→    if (view === "admin_events") {
   646→      const adminAccess = resolveAdminViewAccess(authContext);
   647→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   648→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   649→      const adminEvents = await buildAdminSubmissionEvents(scopedEvents);
   650→      return jsonResponse(adminEvents, { status: 200 });
   651→    }
   652→
   653→    if (view === "review_queue") {
   654→      const adminAccess = resolveAdminViewAccess(authContext);
   655→      if (adminAccess === "unauthorized") return errorResponse("Unauthorized", 401);
   656→      if (adminAccess === "forbidden") return errorResponse("Forbidden", 403);
   657→      const pending = scopedEvents.filter((event) => isPendingReview(event));
   658→      const adminEvents = await buildAdminSubmissionEvents(pending);
   659→      return jsonResponse(adminEvents, { status: 200 });
   660→    }
   661→
   662→    let projected = projectPointsFromEvents(scopedEvents);
   663→
   664→    if (view === "dedup_candidates") {
   665→      if (!authContext) return errorResponse("Unauthorized", 401);
   666→      const category = normalizeCategory(url.searchParams.get("category") ?? undefined);
   667→      if (!category) return errorResponse("Invalid category", 400);
   668→      const latitude = Number(url.searchParams.get("lat"));
   669→      const longitude = Number(url.searchParams.get("lng"));
   670→      if (!Number.isFinite(latitude) || !Number.isFinite(longitude)) {
   671→        return errorResponse("lat and lng are required", 400);
   672→      }
   673→      const name = trimString(url.searchParams.get("name"), 160);
   674→      const probeDetails: SubmissionDetails = {};
   675→      if (name) {
   676→        probeDetails.name = name;
   677→        probeDetails.siteName = name;
   678→        probeDetails.roadName = name;
   679→      }
   680→      const result = buildDedupCandidates(
   681→        category,
   682→        { latitude, longitude },
   683→        probeDetails,
   684→        projected,
   685→      );
   686→      return jsonResponse(result);
   687→    }
   688→
   689→    if (lat && lng && radius) {
   690→      const latitude = Number(lat);
   691→      const longitude = Number(lng);
   692→      const radiusKm = Number(radius);
   693→      if (Number.isFinite(latitude) && Number.isFinite(longitude) && Number.isFinite(radiusKm)) {
   694→        projected = projected.filter((point) => haversineKm(point.location, { latitude, longitude }) <= radiusKm);
   695→      }
   696→    }
   697→
   698→    return jsonResponse(projected, {
   699→      status: 200,
   700→      headers: canUsePublicCache
   701→        ? {
   702→            "cache-control": PUBLIC_READ_CACHE_CONTROL,
   703→          }
   704→        : undefined,
   705→    });
   706→  } catch (error) {
   707→    if (isStorageUnavailableError(error)) {
   708→      return errorResponse("Storage service temporarily unavailable", 503, { code: "storage_unavailable" });
   709→    }
   710→    throw error;
   711→  }
   712→}
