    84→    isLowEnd: raw.isLowEnd === true

   218→function groupEventsByPoint(items: AdminSubmissionEvent[], language: 'en' | 'fr'): GroupedPoint[] {
   219→  const groups = new Map<string, AdminSubmissionEvent[]>();
   220→  for (const item of items) {
   221→    const pid = item.event.pointId;
   222→    const existing = groups.get(pid);
   223→    if (existing) {
   224→      existing.push(item);
   225→    } else {
   226→      groups.set(pid, [item]);
   227→    }
   228→  }
   229→  const result: GroupedPoint[] = [];
   230→  for (const [pointId, events] of groups) {
   231→    const sorted = [...events].sort((a, b) => new Date(a.event.createdAt).getTime() - new Date(b.event.createdAt).getTime());
   232→    const latestEvent = sorted[sorted.length - 1]!;
   233→    const createdEvent = sorted.find((e) => e.event.eventType === 'CREATE_EVENT') ?? null;
   234→    const enrichEvents = sorted.filter((e) => e.event.eventType === 'ENRICH_EVENT');
   235→    const allPhotos: GroupedPoint['allPhotos'] = [];
   236→    for (const ev of sorted) {
   237→      const photoUrl = ev.event.photoUrl;
   238→      if (photoUrl && typeof photoUrl === 'string' && photoUrl.trim()) {
   239→        allPhotos.push({
   240→          url: photoUrl,
   241→          eventType: ev.event.eventType,
   242→          createdAt: ev.event.createdAt,
   243→          metadata: ev.fraudCheck?.primaryPhoto ?? null,
   244→        });
   245→      }
   246→      const details = ev.event.details as SubmissionDetails;
   247→      const secondUrl = typeof details.secondPhotoUrl === 'string' && details.secondPhotoUrl.trim() ? details.secondPhotoUrl : null;
   248→      if (secondUrl) {
   249→        allPhotos.push({
   250→          url: secondUrl,
   251→          eventType: ev.event.eventType + ' (secondary)',
   252→          createdAt: ev.event.createdAt,
   253→          metadata: ev.fraudCheck?.secondaryPhoto ?? null,
   254→        });
   255→      }
   256→    }
   257→    result.push({
   258→      pointId,
   259→      events: sorted,
   260→      category: latestEvent.event.category,
   261→      siteName: getSiteName(createdEvent ?? latestEvent, language),
   262→      latestEvent,
   263→      createdEvent,
   264→      enrichEvents,
   265→      allPhotos,
   266→    });
   267→  }
   268→  return result.sort((a, b) => new Date(b.latestEvent.event.createdAt).getTime() - new Date(a.latestEvent.event.createdAt).getTime());
   269→}
